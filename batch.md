@echo off

rem ------------------------------------
set width=1024
set height=512
set ext=jpg
set src=source

set a=%a% /convert=%cd%\%dst%\*.%ext%

rem ------------------------------------
mkdir %dst%
%iview% %a%
dir /b %dst% > files.txt
