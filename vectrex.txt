==hello world==
<code c>
#include <vectrex/bios.h>

int main() { 
  while(1) { 
    waitretrace(); 
    printstr_c( 0x10, -0x50, "HELLO WORLD!" ); 
  } return 0; 
}
</code>

<code bash>
cmoc --vectrex hello_world.c
</code>
